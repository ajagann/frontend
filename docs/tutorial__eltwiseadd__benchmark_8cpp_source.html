173 hebench::APIBridge::Handle TutorialEltwiseAddBenchmark::encode(const hebench::APIBridge::PackedData *p_parameters)
174 {
175  // To be compatible with decode and the method signature, this encode bundles
176  // multiple data packs inside a single handle. Since in this example we are
177  // doing 1 plaintext parameter and 1 ciphertext parameter, each bundle will
178  // contain a single data pack.
179 
180  assert(p_parameters && p_parameters->pack_count > 0 && p_parameters->p_data_packs);
181 
182  TutorialEngine &engine = reinterpret_cast<TutorialEngine &>(this->getEngine());
183 
185  // bundle multiple data packs together, even though we will have only 1 per call
186  std::vector<InternalParams> params(p_parameters->pack_count);
187 
188  for (std::size_t datapack_i = 0; datapack_i < p_parameters->pack_count; ++datapack_i)
189  {
190  const hebench::APIBridge::DataPack &datapack = p_parameters->p_data_packs[datapack_i];
191  assert(datapack.buffer_count > 0 && datapack.p_buffers);
192 
193  params[datapack_i].samples.resize(datapack.buffer_count);
194  params[datapack_i].param_position = datapack.param_position;
195  params[datapack_i].tag = InternalParams::tagPlaintext;
196  }
198 
200  for (std::size_t datapack_i = 0; datapack_i < p_parameters->pack_count; ++datapack_i)
201  {
202  const hebench::APIBridge::DataPack &datapack = p_parameters->p_data_packs[datapack_i];
203  for (std::uint64_t sample_i = 0; sample_i < datapack.buffer_count; ++sample_i)
204  {
205  const hebench::APIBridge::NativeDataBuffer &sample_buffer =
206  datapack.p_buffers[sample_i];
207  assert(sample_buffer.p && sample_buffer.size / sizeof(double) > 0);
208  // read from raw data: the data type is Float64 as specified in this benchmark description
209  gsl::span<const double> clear_text =
210  gsl::span<const double>(reinterpret_cast<const double *>(sample_buffer.p),
211  sample_buffer.size / sizeof(double));
212 
213  seal::Plaintext encoded;
214  engine.ckksEncoder().encode(clear_text, engine.scale(), encoded);
215  // store the encoded plaintext in the parameter samples
216  params[datapack_i].samples[sample_i] = std::make_shared<seal::Plaintext>(std::move(encoded));
217  }
218  }
220 
222  // wrap our internal object into a handle to cross the boundary of the API Bridge
223  return engine.template createHandle<decltype(params)>(sizeof(seal::Plaintext) * params.size(), // size (arbitrary for our usage if we need to)
224  InternalParams::tagPlaintext, // extra tags
225  std::move(params)); // constructor parameters
227 }

231 void TutorialEltwiseAddBenchmark::decode(hebench::APIBridge::Handle h_encoded_data, hebench::APIBridge::PackedData *p_native)
232 {
233  // This decode is able to decode multiple data packs bundled in a single handle
234 
235  assert(p_native && p_native->p_data_packs && p_native->pack_count > 0);
236 
237  TutorialEngine &engine = reinterpret_cast<TutorialEngine &>(this->getEngine());
238  // This method should handle decoding of data encoded using encode(), due to
239  // specification stating that encode() and decode() are inverses; as well as
240  // handle data decrypted from operation() results.
241 
242  // retrieve our internal format object from the handle
243  const std::vector<InternalParams> &encoded =
244  engine.template retrieveFromHandle<std::vector<InternalParams>>(h_encoded_data);
245 
246  // according to specification, we must decode as much data as possible, where
247  // any excess encoded data that won't fit into the pre-allocated native buffer
248  // shall be ignored
249 
250  std::uint64_t min_datapack_count = std::min(p_native->pack_count, encoded.size());
251  for (std::size_t datapack_i = 0; datapack_i < min_datapack_count; ++datapack_i)
252  {
253  // decode the next data pack
254  hebench::APIBridge::DataPack *p_native_datapack = &p_native->p_data_packs[datapack_i];
255 
256  // find the encoded data pack corresponding to the requested clear text data pack
257  const InternalParams *p_encoded_datapack = nullptr;
258  for (std::size_t encoded_i = 0; !p_encoded_datapack && encoded_i < encoded.size(); ++encoded_i)
259  if (encoded[encoded_i].param_position == p_native_datapack->param_position)
260  p_encoded_datapack = &encoded[encoded_i];
261 
262  if ((p_encoded_datapack->tag & InternalParams::tagPlaintext) != InternalParams::tagPlaintext)
263  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS("Invalid tag detected for handle 'h_encoded_data'."),
264  HEBENCH_ECODE_INVALID_ARGS);
265 
266  if (p_native_datapack && p_native_datapack->buffer_count > 0)
267  {
268  std::uint64_t min_sample_count = std::min(p_native_datapack->buffer_count, encoded[datapack_i].samples.size());
269  for (std::uint64_t sample_i = 0; sample_i < min_sample_count; ++sample_i)
270  {
271  // alias the samples
272  hebench::APIBridge::NativeDataBuffer &native_sample = p_native_datapack->p_buffers[sample_i];
273  seal::Plaintext &encoded_sample = *reinterpret_cast<seal::Plaintext *>(p_encoded_datapack->samples[sample_i].get());
274 
275  // decode as much as possible
276  std::vector<double> decoded;
277  engine.ckksEncoder().decode(encoded_sample, decoded);
278  std::copy_n(decoded.begin(),
279  std::min(decoded.size(), native_sample.size / sizeof(double)),
280  reinterpret_cast<double *>(native_sample.p));
281  }
282  }
283  }

399 void TutorialEltwiseAddBenchmark::store(hebench::APIBridge::Handle h_remote_data,
400  hebench::APIBridge::Handle *p_local_data, std::uint64_t count)
401 {
402  if (count > 0 && !p_local_data)
403  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS("Invalid null array of handles: \"p_local_data\""),
404  HEBENCH_ECODE_INVALID_ARGS);
405 
406  if (count > 0)
407  {
408  std::vector<InternalParams> plain_data;
409  std::vector<InternalParams> encrypted_data;
410  const std::vector<InternalParams> &remote_data =
411  this->getEngine().template retrieveFromHandle<std::vector<InternalParams>>(h_remote_data);
412 
413  // since remote and host are the same for this example, we just need to return a copy
414  // of the remote as local data.
415  for (const auto &internal_params : remote_data)
416  {
417  if ((internal_params.tag & InternalParams::tagCiphertext) == InternalParams::tagCiphertext)
418  encrypted_data.push_back(internal_params);
419  else if ((internal_params.tag & InternalParams::tagPlaintext) == InternalParams::tagPlaintext)
420  plain_data.push_back(internal_params);
421  else
422  {
423  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS("Unknown tag detected in data pack"),
424  HEBENCH_ECODE_INVALID_ARGS);
425  }
426  }
427  if (!encrypted_data.empty())
428  // store encrypted data in first handle, as per docs
429  p_local_data[0] = this->getEngine().template createHandle<decltype(encrypted_data)>(encrypted_data.size(),
430  InternalParams::tagCiphertext,
431  std::move(encrypted_data));
432  if (!plain_data.empty())
433  {
434  // store plain data in next available handle as per docs
435  hebench::APIBridge::Handle *p_h_plain = nullptr;
436  if (encrypted_data.empty())
437  p_h_plain = &p_local_data[0];
438  else if (count > 1)
439  p_h_plain = &p_local_data[1];
440 
441  if (p_h_plain)
442  *p_h_plain = this->getEngine().template createHandle<decltype(plain_data)>(plain_data.size(),
443  InternalParams::tagPlaintext,
444  std::move(plain_data));
445  }
446  }
447 
448  // pad with zeros any remaining local handles as per specifications
449  for (std::uint64_t i = 2; i < count; ++i)
450  std::memset(p_local_data + i, 0, sizeof(hebench::APIBridge::Handle));
451 }

455 hebench::APIBridge::Handle TutorialEltwiseAddBenchmark::operate(hebench::APIBridge::Handle h_remote_packed,
456  const hebench::APIBridge::ParameterIndexer *p_param_indexers)
457 {
458  TutorialEngine &engine = reinterpret_cast<TutorialEngine &>(this->getEngine());
459 
461  if ((h_remote_packed.tag & (InternalParams::tagCiphertext | InternalParams::tagPlaintext)) != (InternalParams::tagCiphertext | InternalParams::tagPlaintext))
462  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS("Invalid tag detected for handle 'h_remote_packed'."),
463  HEBENCH_ECODE_INVALID_ARGS);
464  const std::vector<InternalParams> &loaded_data =
465  engine.template retrieveFromHandle<std::vector<InternalParams>>(h_remote_packed);
466 
467  assert(loaded_data.size() == ParametersCount);
469 
471  // retrieve the plaintext parameter and the ciphertext parameter
472  const std::vector<std::shared_ptr<void>> *p_params[ParametersCount] = { nullptr, nullptr };
473 
474  for (std::size_t i = 0; i < loaded_data.size(); ++i)
475  {
476  if (loaded_data[i].param_position == 0
477  && (loaded_data[i].tag & InternalParams::tagPlaintext) == InternalParams::tagPlaintext)
478  p_params[0] = &loaded_data[i].samples; // param 0 is plain text
479  else if (loaded_data[i].param_position == 1
480  && (loaded_data[i].tag & InternalParams::tagCiphertext) == InternalParams::tagCiphertext)
481  p_params[1] = &loaded_data[i].samples; // param 1 is ciphertext
482  }
483 
484  // validate extracted parameters
485  for (std::size_t i = 0; i < ParametersCount; ++i)
486  {
487  if (!p_params[i])
488  {
489  std::stringstream ss;
490  ss << "Unable to find operation parameter " << i << " loaded.";
491  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS(ss.str()),
492  HEBENCH_ECODE_INVALID_ARGS);
493  }
494  }
496 
498  std::uint64_t results_count = 1;
499  for (std::size_t param_i = 0; param_i < ParametersCount; ++param_i)
500  {
501  if (p_param_indexers[param_i].value_index >= p_params[param_i]->size())
502  {
503  std::stringstream ss;
504  ss << "Invalid parameter indexer for operation parameter " << param_i << ". Expected index in range [0, "
505  << p_params[param_i]->size() << "), but " << p_param_indexers[param_i].value_index << " received.";
506  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS(ss.str()),
507  HEBENCH_ECODE_INVALID_ARGS);
508  }
509  else if (p_param_indexers[param_i].value_index + p_param_indexers[param_i].batch_size > p_params[param_i]->size())
510  {
511  std::stringstream ss;
512  ss << "Invalid parameter indexer for operation parameter " << param_i << ". Expected batch size in range [1, "
513  << p_params[param_i]->size() - p_param_indexers[param_i].value_index << "], but " << p_param_indexers[param_i].batch_size << " received.";
514  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS(ss.str()),
515  HEBENCH_ECODE_INVALID_ARGS);
516  }
517  results_count *= p_param_indexers[param_i].batch_size; // count the number of results expected
518  }
520 
522  // allocate space for results
523  std::vector<InternalParams> results(ResultComponentsCount);
524  results.front().samples.resize(results_count); // ResultComponentsCount == 1 for this workload
525  results.front().param_position = 0; // result component
526  results.front().tag = InternalParams::tagCiphertext;
528 
530  // perform the actual operation
531  // keep in mind the result ordering (for offline test, most significant parameter moves faster)
532  std::size_t result_i = 0;
533  for (std::uint64_t p0_sample_i = p_param_indexers[0].value_index;
534  p0_sample_i < p_param_indexers[0].value_index + p_param_indexers[0].batch_size;
535  ++p0_sample_i)
536  {
537  for (std::uint64_t p1_sample_i = p_param_indexers[1].value_index;
538  p1_sample_i < p_param_indexers[1].value_index + p_param_indexers[1].batch_size;
539  ++p1_sample_i)
540  {
541  const seal::Plaintext &p0 =
542  *reinterpret_cast<const seal::Plaintext *>(p_params[0]->at(p0_sample_i).get());
543  const seal::Ciphertext &p1 =
544  *reinterpret_cast<const seal::Ciphertext *>(p_params[1]->at(p1_sample_i).get());
545  seal::Ciphertext result;
546  engine.evaluator().add_plain(p1, p0, result);
547 
548  // store result in our internal representation
549  results.front().samples[result_i] = std::make_shared<seal::Ciphertext>(std::move(result));
550  ++result_i;
551  }
552  }
554 
555  // send our internal result across the boundary of the API Bridge as a handle
556  return engine.template createHandle<decltype(results)>(sizeof(seal::Ciphertext) * results_count,
