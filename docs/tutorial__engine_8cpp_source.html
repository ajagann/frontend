63 void TutorialEngine::initCKKS(std::size_t poly_modulus_degree, std::size_t num_coeff_moduli, int scale_exponent)
64 {
65  if (num_coeff_moduli < 2)
66  throw hebench::cpp::HEBenchError(HEBERROR_MSG_CLASS("Invalid CKKS initialization parameters. There must be, at least, 2 coefficient moduli: \"num_coeff_moduli\"."),
67  HEBENCH_ECODE_INVALID_ARGS);
68 
69  std::vector<int> coeff_modulus = { 60 };
70  double scale = pow(2.0, scale_exponent);
71  for (std::size_t i = 0; i < num_coeff_moduli - 1; ++i)
72  coeff_modulus.push_back(scale_exponent);
73  coeff_modulus.push_back(60);
74  // coeff_modulus = { 60, 40, 40, ..., 60 }
75  // coeff_modulus.size() == num_coeff_moduli + 1
76 
77  m_scheme = seal::scheme_type::CKKS;
78  seal::EncryptionParameters parameters(seal::scheme_type::CKKS);
79  parameters.set_poly_modulus_degree(poly_modulus_degree);
80  parameters.set_coeff_modulus(seal::CoeffModulus::Create(poly_modulus_degree, coeff_modulus));
81  m_context = seal::SEALContext::Create(parameters);
82  m_scale = scale;
83 
84  m_keygen = std::unique_ptr<seal::KeyGenerator>(new seal::KeyGenerator(m_context));
85  m_public_key = std::unique_ptr<seal::PublicKey>(new seal::PublicKey(m_keygen->public_key()));
86  m_secret_key = std::unique_ptr<seal::SecretKey>(new seal::SecretKey(m_keygen->secret_key()));
87  m_relin_keys = std::unique_ptr<seal::RelinKeys>(new seal::RelinKeys(m_keygen->relin_keys_local()));
88  m_encryptor = std::unique_ptr<seal::Encryptor>(new seal::Encryptor(m_context, *m_public_key));
89  m_evaluator = std::unique_ptr<seal::Evaluator>(new seal::Evaluator(m_context));
90  m_decryptor = std::unique_ptr<seal::Decryptor>(new seal::Decryptor(m_context, *m_secret_key));
91  m_ckks_encoder = std::unique_ptr<seal::CKKSEncoder>(new seal::CKKSEncoder(m_context));
92  m_galois_keys = std::unique_ptr<seal::GaloisKeys>(new seal::GaloisKeys(m_keygen->galois_keys_local()));
93 }
